class Solution {
    public int longestConsecutive(int[] a) {
        Set<Integer> set = new HashSet<>();
        for (int i=0;i<a.length;i++){
            set.add(a[i]);
        }

        int max=0;
        for (int x : set){
            if (!set.contains(x-1)){
                int count = 1;
                while (set.contains(x+count)){
                    count++;
                }
                max = Math.max(max, count);
            }
        }
        return max;
    }
}
